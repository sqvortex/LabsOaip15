	S = sqrt(D);
	printf("Корень дискриминанта равен = %f\n", S);
	
	


	if (D == 0 && a > 0)
	{
		printf("Нет решений!");
	}

		x1 = (-b + S) / (2 * a);
		printf("Решение: (-Бесконечности; +Бесконечности)U{x1} = %f", x1);
	}
	if (D > 0 && a > 0)
	{
		x1 = (-b + S) / (2 * a);
		x2 = (-b - S) / (2 * a);
		printf("x1= %f\n", x1);
		printf("x2= %f ", x2);
	}
	if (D > 0 && a < 0)
	{
		x1 = (-b + S) / (2 * a);
		x2 = (-b - S) / (2 * a);
		printf("x1= %f\n", x1);
		printf("x2= %f\n ", x2);
		printf("Решение: (-Бескон; %f)U(%f; +Бескон)", x1, x2);
	}
	if (D < 0 && a < 0)
	{
		printf(" Решение: Все значения!");
	}
	if (a == 0)
	{
		double x = 0;

	}
}
